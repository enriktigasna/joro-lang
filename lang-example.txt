; Factorial calculation in simple language

; Function to calculate factorial of a number
(defn factorial (n) (
    (var result = 1)
    (while (n > 1) (
        (set result = (result * n))
        (set n = (n - 1))
    ))
    (return result)
))

; Main block
(main (
    (println ("Enter a number:"))
    (var number = (input)) ; Assuming input is a function that gets user input
    (var result = (factorial number))
    (println ("Factorial of " + number + " is " + result))
))
